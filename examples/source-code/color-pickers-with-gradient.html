  <div class="explanation">
    <p>This example combines use of two <a href="../reference/Uize.Widget.ColorPicker.html"><code>Uize.Widget.ColorPicker</code></a> instances with a single <a href="../reference/Uize.Widget.ColorCube.html"><code>Uize.Widget.ColorCube</code></a> instance to display a gradient between two colors you select. The color cube is turned into a simple gradient display by setting the value of its <code>divisionsY</code> state property to <code>1</code>, and then using only the <code>colorTopLeft</code> and <code>colorTopRight</code> state properties to change the gradient's color poles.</p>

<script type="text/javascript">

Uize.module ({
  required:[
    'UizeSite.Page.Example.library',
    'UizeSite.Page.Example',
    'Uize.Widget.ColorPicker',
    'Uize.Widget.ColorCube'
  ],
  builder:function () {
    /*** create the example page widget ***/
      var page = window.page = UizeSite.Page.Example ();

    /*** add the color cube child widget used for the gradient effect ***/
      var colorCube = page.addChild (
        'colorCube',
        Uize.Widget.ColorCube,
        {
          divisionsX:100,
          divisionsY:1,
          colorTopLeft:'ffff00',
          colorTopRight:'ff0000',
          built:false
        }
      );

    /*** add the color picker child widgets ***/
      function setupColorPicker (colorName,initialValue) {
        var colorPicker = page.addChild (
          colorName + 'Picker',
          Uize.Widget.ColorPicker,
          {
            value:colorCube.get (colorName),
            built:false
          }
        );
        colorPicker.wire ('Changed.value',function () {colorCube.set (colorName,colorPicker + '')});
      }
      setupColorPicker ('colorTopLeft');
      setupColorPicker ('colorTopRight');

    /*** wire up the page widget ***/
      page.wireUi ();
  }
});

</script>
