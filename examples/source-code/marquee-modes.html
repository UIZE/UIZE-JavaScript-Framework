<!-- JavaScript code to make the static marquee HTML "come alive" -->

<script type="text/javascript">

Uize.module ({
  required:[
    'UizeDotCom.Page.Example.library',
    'UizeDotCom.Page.Example',
    'Uize.Widget.Resizer.Marquee'
  ],
  builder:function () {
    /*** create the example page widget ***/
      var page = window.page = new UizeDotCom.Page.Example;

    /*** create the marquee instances ***/
      page.addChild (
        'marquee0',
        Uize.Widget.Resizer.Marquee,
        {
          aspectRatio:2,
          constrain:true,
          width:100,
          height:50,
          minWidth:100,
          minHeight:50,
          built:false
        }
      );
      page.addChild (
        'marquee1',
        Uize.Widget.Resizer.Marquee,
        {
          aspectRatio:1,
          constrain:true,
          width:50,
          height:50,
          minWidth:20,
          minHeight:20,
          built:false
        }
      );
      page.addChild (
        'marquee2',
        Uize.Widget.Resizer.Marquee,
        {
          constrain:true,
          width:100,
          height:50,
          minWidth:20,
          minHeight:20,
          built:false
        }
      );
      page.addChild (
        'marquee3',
        Uize.Widget.Resizer.Marquee,
        {
          constrain:false,
          width:100,
          height:50,
          minWidth:20,
          minHeight:20,
          built:false
        }
      );
      page.addChild (
        'marquee4',
        Uize.Widget.Resizer.Marquee,
        {
          constrain:true,
          width:100,
          height:50,
          minWidth:20,
          minHeight:20,
          fixedX:true,
          built:false
        }
      );
      page.addChild (
        'marquee5',
        Uize.Widget.Resizer.Marquee,
        {
          constrain:true,
          width:100,
          height:50,
          minWidth:20,
          minHeight:20,
          fixedX:true,
          fixedY:true,
          built:false
        }
      );

    /*** wire up the page widget ***/
      page.wireUi ();
  }
});

</script>
