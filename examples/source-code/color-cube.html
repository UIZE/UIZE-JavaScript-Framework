  <div class="explanation">
    <p>In this example, an instance of the <a href="../reference/Uize.Widget.ColorCube.html"><code>Uize.Widget.ColorCube</code></a> class is wired up to provide a color cube that shows a color matrix (or color grid) of all the interpolated colors between four adjustable corner colors. Enter hex RGB values for any colors at any of the four corners, then click the "UPDATE" button below the cube to see the result. This cube is 10x10 and represents 100 colors, but the widget can be configured for different numbers of divisions on both the X and Y axes by setting the <code>divisionsX</code> and <code>divisionsY</code> set-get properties, respectively. With additional code, a subclass or other widget could use the color cube widget as a color selection palette, allowing the user to explore color ranges in 2D.</p>

<script type="text/javascript">

Uize.module ({
  required:[
    'UizeDotCom.Page.Example.library',
    'UizeDotCom.Page.Example',
    'Uize.Widget.ColorCube'
  ],
  builder:function () {
    /*** create the example page widget ***/
      var page = window.page = new UizeDotCom.Page.Example;

    /*** create the Uize.Widget.ColorCube object ***/
      var colorCube = page.addChild ('colorCube',Uize.Widget.ColorCube,{built:false});

      function updateColorCubeFromInputs () {
        colorCube.set ({colorTopLeft:page.getNodeValue ('colorTopLeft')});
        colorCube.set ({colorTopRight:page.getNodeValue ('colorTopRight')});
        colorCube.set ({colorBottomLeft:page.getNodeValue ('colorBottomLeft')});
        colorCube.set ({colorBottomRight:page.getNodeValue ('colorBottomRight')});
      }

    /*** initialize input nodes ***/
      page.setNodeValue ('colorTopLeft',colorCube.get ('colorTopLeft'));
      page.setNodeValue ('colorTopRight',colorCube.get ('colorTopRight'));
      page.setNodeValue ('colorBottomLeft',colorCube.get ('colorBottomLeft'));
      page.setNodeValue ('colorBottomRight',colorCube.get ('colorBottomRight'));

    /*** wire up the page widget ***/
      page.wireUi ();

    /*** wire update button ***/
      page.wireNode ('updateButton','click',updateColorCubeFromInputs);
  }
});

</script>
