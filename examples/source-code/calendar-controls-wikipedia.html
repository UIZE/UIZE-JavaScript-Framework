  <div class="explanation">
    <p>In this example, an instance of the <a href="../reference/Uize.Widget.Calendar.html"><code>Uize.Widget.Calendar</code></a> class is used to wire up a simple calendar widget that controls the URL for a document in an IFRAME to its right. Use the calendar to choose a date and the IFRAME will load information about what happened on that day in time, courtesy of <a href="http://www.wikipedia.org" target="_blank">wikipedia.org</a>. As powerful of a resource as it is, Wikipedia does not - unfortunately - contain reference information for the future. Because of this failing on Wikipedia's part, the calendar widget is configured to not allow selection of dates in the future, through the use of its <code>maxValue</code> state property.</p>

<!-- JavaScript code to insert the calendar widget and wire up the page -->

<script type="text/javascript">

Uize.module ({
  required:[
    'UizeSite.Page.Example.library',
    'UizeSite.Page.Example',
    'Uize.Widget.Calendar',
    'Uize.Templates.Calendar',
    'Uize.Date'
  ],
  builder:function () {
    /*** create the example page widget ***/
      var page = window.page = UizeSite.Page.Example ();

    /*** update wikipedia iframe to new date ***/
      function updateWikipediaIframe () {
        var
          date = calendar.valueOf (),
          monthName = Uize.Date.monthNames [date.getMonth ()],
          fullYear = date.getFullYear ()
        ;
        page.setNodeProperties (
          'wikipediaIframe',
          {
            src:
              'http://www.wikipedia.org/wiki/' +
              monthName + '_' + fullYear + '#' + fullYear + '_' + monthName + '_' + date.getDate ()
          }
        );
      }

    /*** add the calendar child widget ***/
      var calendar = page.addChild (
        'calendar',
        Uize.Widget.Calendar,
        {
          built:false,
          html:Uize.Templates.Calendar,
          maxValue:new Date
        }
      );
      calendar.wire ('Changed.value',updateWikipediaIframe);

    /*** wire up the page widget ***/
      page.wireUi ();

    /*** initialize the wikipedia iframe ***/
      updateWikipediaIframe ();
  }
});

</script>
