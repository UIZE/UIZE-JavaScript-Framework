      var _date = new Date (
        +_year + (_year < 100 && 400),
        (+_month || 1) - 1,

        +_hours || 0,
        +_minutes || 0,
        +_seconds || 0,
        +_milliseconds || 0
      );
      _year < 100 && _date.setFullYear (_year);
      return _date;
